/**
 * 🔒 스피킹 트레이너 웹 앱 API (Server Side)
 * - Vercel(또는 외부 웹앱)에서 fetch()로 데이터를 요청받아 JSON 반환 담당
 */

// ▼▼▼ [설정] 카페 공지사항에 올릴 비밀번호를 여기서 정하세요 ▼▼▼
const APP_PASSWORD = "stella111"; 
// ▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲▲

/**
 * 외부(Vercel 등)에서 GET 요청을 받을 때 실행되는 함수
 * 예시 호출 URL: https://script.google.com/macros/s/.../exec?password=stella369
 */
function doGet(e) {
  // CORS 프리플라이트를 피하기 위한 범용적인 헤더 및 JSON 응답 반환
  
  // 1. 파라미터에서 비밀번호 추출
  const inputPassword = e.parameter.password;
  
  // 2. 비밀번호 검사
  if (inputPassword !== APP_PASSWORD) {
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: "비밀번호가 틀렸습니다. 카페 공지사항을 확인하세요."
    })).setMimeType(ContentService.MimeType.JSON);
  }

  try {
    // 3. 모든 시트(탭) 데이터 가져오기
    const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
    const sheets = spreadsheet.getSheets();
    
    let allData = [];
    let globalIndex = 0; // 고유 번호 부여를 위한 인덱스
    
    for (let i = 0; i < sheets.length; i++) {
        const sheet = sheets[i];
        const sheetName = sheet.getName();
        const lastRow = sheet.getLastRow();
        
        // 데이터가 없는 시트는 건너뜀
        if (lastRow < 5) continue;
        
        const rowsCount = lastRow - 4;
        
        // A열부터 8칸(H열) 평문 가져오기 (기존 호환)
        const rangeValues = sheet.getRange(5, 1, rowsCount, 8).getValues();
        // H열(영어)의 텍스트 색상 서식만 별도로 파싱하기 위해 RichText 객체 가져오기
        const richTextRange = sheet.getRange(5, 8, rowsCount, 1).getRichTextValues();
        
        const sheetData = [];
        for (let r = 0; r < rowsCount; r++) {
            const row = rangeValues[r];
            const category = row[0];
            const korean = row[2] ? String(row[2]) : "";
            const englishPlain = row[7] ? String(row[7]).trim() : "";
            
            // 데이터가 없는 빈 줄은 스킵
            if (englishPlain === "" || korean === "") continue;

            const richTextObj = richTextRange[r][0];
            let englishHtml = "";
            
            if (richTextObj && richTextObj.getRuns) { // RichTextValue 방어코드
                const runs = richTextObj.getRuns();
                runs.forEach(run => {
                    const textChunk = run.getText();
                    if (!textChunk) return;
                    
                    const textStyle = run.getTextStyle();
                    const color = textStyle ? textStyle.getForegroundColor() : null;
                    
                    // 시트 작성 시 포함된 특수문자 HTML 자동 이스케이프
                    const safeText = textChunk.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
                    
                    // 검은색(#000000)이 아니면 색상 span 렌더링 유지
                    if (color && color !== "#000000") {
                        englishHtml += `<span style="color:${color}; font-weight:bold;">${safeText}</span>`;
                    } else {
                        englishHtml += safeText;
                    }
                });
            } else {
                englishHtml = englishPlain.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
            }
            
            sheetData.push({
                id: globalIndex++, // 각 카드 고유 번호
                subject: sheetName, // 추가됨: 탭 이름 (과목명)
                category: category,   // 챕터명
                korean: korean,     // 한국어
                english: englishPlain, // 기존 호환: 평문(TTS 및 검색결과용)
                englishHtml: englishHtml // 전용: 서식이 포함된 html (화면 렌더링용)
            });
        }
        
        allData = allData.concat(sheetData);
    }

    // 4. 결합된 데이터를 JSON 포맷으로 변환하여 반환
    return ContentService.createTextOutput(JSON.stringify({ status: "success", data: allData }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    // 에러 발생 시 처리
    return ContentService.createTextOutput(JSON.stringify({
      status: "error",
      message: "구글 시트 데이터를 가져오는 중 오류가 발생했습니다: " + error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}
